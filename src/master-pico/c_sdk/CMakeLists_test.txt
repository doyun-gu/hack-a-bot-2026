cmake_minimum_required(VERSION 3.13)

# Auto-detect board: set PICO_BOARD=pico or PICO_BOARD=pico2
# Pass -DPICO_BOARD=pico for RP2040, -DPICO_BOARD=pico2 for RP2350

include($ENV{PICO_SDK_PATH}/external/pico_sdk_import.cmake)

project(gridbox_test C CXX ASM)
set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)

pico_sdk_init()

add_executable(gridbox_test test_blink.c)

target_link_libraries(gridbox_test pico_stdlib hardware_adc)

pico_enable_stdio_usb(gridbox_test 1)
pico_enable_stdio_uart(gridbox_test 0)

pico_add_extra_outputs(gridbox_test)
